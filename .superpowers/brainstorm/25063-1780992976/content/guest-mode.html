Random Mission Logic

// lib/ai/guest-mission.ts // Picks a random mission for guest users each day. // Deterministic per date so refreshing gives the same mission. const GUEST_MISSIONS = [ { year: "8", missionId: "missions/percentages" }, { year: "8", missionId: "missions/ratio" }, { year: "8", missionId: "missions/algebra-basics" }, { year: "8", missionId: "missions/graphs" }, { year: "8", missionId: "missions/angles" }, { year: "9", missionId: "missions/simultaneous-equations" }, { year: "9", missionId: "missions/pythagoras" }, { year: "9", missionId: "missions/indices" }, { year: "9", missionId: "missions/probability" }, { year: "8", missionId: "missions/fractions" }, ]; function pickRandomMission(): MissionDef { const seed = dateSeed(new Date()); // e.g. "2026-06-09" const idx = hash(seed) % GUEST_MISSIONS.length; return loadMission(GUEST_MISSIONS[idx].missionId); }
